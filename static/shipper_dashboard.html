<script>

async function createJob() {


    const job = {

        pickup_location: document.getElementById("pickup").value,

        dropoff_location: document.getElementById("dropoff").value,

        load_description: document.getElementById("description").value,

        weight: document.getElementById("weight").value,

        equipment_type: document.getElementById("equipment").value,

        delivery_window: document.getElementById("delivery").value,

        contact_phone: document.getElementById("phone").value,

        price_offered: parseFloat(document.getElementById("price").value)

    };


    const messageBox = document.getElementById("message");


    try {

        const response = await fetch("https://YOUR_RAILWAY_URL_HERE/jobs", {

            method: "POST",

            headers: { "Content-Type": "application/json" },

            body: JSON.stringify(job)

        });


        const result = await response.json();


        if (response.ok) {

            messageBox.style.display = "block";

            messageBox.className = "message success";

            messageBox.innerText = "Job created successfully! ID: " + result.id;


            document.querySelectorAll("input, textarea, select").forEach(el => el.value = "");

        } else {

            throw new Error(result.detail || "Failed to create job");

        }

    } catch (err) {

        messageBox.style.display = "block";

        messageBox.className = "message error";

        messageBox.innerText = "Error: " + err.message;

    }

}

</script>
